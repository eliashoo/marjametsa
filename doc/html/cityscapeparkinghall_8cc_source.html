80 // int CityScapeParkingHall::getX() const
81 // ============================================================================
82 // Palauttaa parkkihallin x-karttakoordinaatin.
83 //
84 // Ei parametrejä.
85 //
86 // Paluuarvot:
87 // (1) Parkkihallin x-koordinaatti.
88 // ============================================================================
89 int CityScapeParkingHall::getX() const
90 {
91  // Google-koordinaatti kartan vasemmalle reunalle * 10^6
92  int mapLeftEdge = 23744979;
93 
94  // Oikean ja vasemman reunan välinen erotus jaettuna karttakuvan leveydellä
95  // (23789261 - 23744979) / 1024;
96  double scaleCoefficient = 43.244140625;
97 
98  // Palautetaan käännös karttakoordinaateista pikseleihin
99  // Käännös on summittainen, eikä ota huomioon maan pinnan pyöreyttä.
100  // Näin pienellä alueella tulos kuitenkin ok.
101  return static_cast<int>( ((longtitude_ - mapLeftEdge) / scaleCoefficient) );
102 }

111 // Paluuarvot:
112 // (1) Parkkihallin y-koordinaatti.
113 // ============================================================================
114 int CityScapeParkingHall::getY() const
115 {
116  // Google-koordinaatti kartan yläreunalle * 10^6
117  int mapUpperEdge = 61506517;
118 
119  // Ylä- ja alareunojen välinen erotus jaettuna karttakuvan korkeudella
120  // (61506517 - 61492238) / 768;
121  double scaleCoefficient = 18.59244791666667;
122 
123  // Palautetaan käännös karttakoordinaateista pikseleihin

125  // Näin pienellä alueella tulos kuitenkin ok.
126  return static_cast<int>( ((mapUpperEdge - latitude_) / scaleCoefficient ) );
