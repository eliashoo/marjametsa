577 int ActionScene::checkShotCollision(QGraphicsLineItem *collLine, uint shooterId, int round)
578 {
579  ActionSceneNPC *hitEnemy(nullptr);
580 
581  bool enemyHit = false;
582  uint hitEnemyId{0};
583  uint shooterSlot{0};
584 
585  // Ensin tarkistetaan osumat esteisiin, ettei ampumalinjan päällekkäisyyttä
586  // esteiden takana mahdollisesti oleviin vihollisiin huomioitaisi.
587  for( uint tile = 0; tile < tilesNear_.at(shooterId).size(); ++tile ) {
588 
589  if( collLine->collidesWithItem(tilesNear_.at(shooterId).at(tile)) ) {
590 
591  QLineF newLine = collLine->line();
592 
593  // Jos ampumalinja osuu esteeseen, lyhennetään sitä.
594  newLine.setLength(newLine.length()-7);
595  collLine->setLine(newLine);
596 
597  // Jos onnistutaan ammuskelemaan seinän sisässä.
598  if( newLine.length() > 10 ) {
599 
600  // Kutsutaan tätä funktiota uudestaan lyhyemmällä linjalla.
601  checkShotCollision(collLine, shooterId);
602  }
603 
604  break;
605  }
606  }
607 
608  // Tarkistetaan osumat vihollisiin.
609  for( uint enemy = 0; enemy < enemies_.size(); ++enemy ) {
610 
611  if( collLine->collidesWithItem(enemies_.at(enemy)) && !enemies_.at(enemy)->isDead() ) {
612 
613  enemyHit = true;
614  hitEnemy = enemies_.at(enemy);
615  hitEnemyId = hitEnemy->getId();
616  shooterSlot = playerParty_.at(shooterId)->getSlot();
617 
618  QLineF newLine = collLine->line();
619 
620  // Jos löydettiin vihollinen ampumalinjalta, lyhennetään linjaa.
621  newLine.setLength(newLine.length()-7);
622  collLine->setLine(newLine);
623 
624  // Jos onnistutaan ammuskelemaan vihollisen päällä.
625  if( newLine.length() > 10 ) {
626 
627  // Jos seuraavalla kierroksella ei enää löydetä osumaa
628  // (palauttaa 0), pidetään nykyinen.
629  if( checkShotCollision(collLine, shooterId, round+1) == 0 &&
630  !hitEnemy->isDead() ) {
631 
632  emit enemyHitSignal(hitEnemyId, shooterId, shooterSlot);
633 
634  // Viholliset käyvät päälle jos niitä ampuu.
635  if( enemies_.at(hitEnemyId)->getTarget() == 5 ) {
636  enemies_.at(hitEnemyId)->setTarget(shooterId);
637  enemies_.at(hitEnemyId)->setAIMode("attack");
638  }
639  }
640  }
641  break;
642  }
643  }
644 
645  // Kun ei löydetä, tai ei enää löydetä osumaa viholliseen palautetaan 0.
646  if( !enemyHit ) {
647  return 0;
648  } else {
649  return round;
650  }
651 }
