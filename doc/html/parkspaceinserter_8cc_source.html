18 void parkSpaceInserter::addParkingSpaces_()
19 {
20  auto groundEdgeTiles = getWallTiles();
21  // Haetaan ja merkitään aloitustile
22  std::uniform_int_distribution<int> entry(0,groundEdgeTiles.size()-1);
23 
24  int entryTile = groundEdgeTiles[entry(map_->rng)];
25  auto end = groundEdgeTiles.end();
26  int wallTile = map_->getClosestEdge(map_->indexToX(entryTile),map_->indexToY(entryTile));
27  map_->tiles_[wallTile] = TileType::ENTRY;
28 
29  for(auto tile = groundEdgeTiles.begin(); tile != end; tile++)
30  {
31  if(*tile == entryTile)
32  {
33  groundEdgeTiles.erase(tile);
34  break;
35  }
36  }
37 
38  int midX = map_->indexToX(entryTile);
39  int midY = map_->indexToY(entryTile);
40  map_->startX = midX;
41  map_->startY = midY;
42 
43  map_->tiles_[ map_->indexAt(midX, midY) ] = TileType::ACCESSIBLE;
44  // Merkkaus valmis
45 
46  // Parkkiruutu koostuu kolmesta tilestä, joista kahteen pysäköidään
47  // ja kolmannella varmistetaan, että ruudusta pääsee pois
48  struct ParkingSpace {
49  int x1; //Ensimmäinen ruutu
50  int y1;
51  int x2; //Toinen ruutu
52  int y2;
53  int nextX; //kolmas ruutu
54  int nextY;
55  int dir; //Seinän suunta parkkiruudusta nähden
56  ParkingSpace(int a, int b, int c, int d, int e, int f,int g): x1(a),y1(b),x2(c),y2(d),nextX(e),nextY(f),dir(g) {}
57  };
58  std::vector<struct ParkingSpace> spaces;
59 
60  std::function<bool(int,int)> tileOk = [=](int x,int y) { return map_->tiles_[map_->indexAt(x,y)] == TileType::FLOOR; };
61 
62  // Sijoitetaan ruudut kartalla maan reunoille
63  for(const auto &t: groundEdgeTiles)
64  {
65  int tileX = map_->indexToX(t);
66  int tileY = map_->indexToY(t);
67 
68  int edge = map_->getClosestEdge(map_->indexToX(t),map_->indexToY(t),TileType::WALL);
69 
70  // Jos seinää ei löytynyt
71  if(edge == -1)
72  {
73  continue;
74  }
75  int edgeX = map_->indexToX(edge);
76  int edgeY = map_->indexToY(edge);
77 
78  // Käydään ruudun ympäristö läpi ja varmistetaan että siinä on tilaa
79 
80  if( tileY == edgeY) // Seinä idässä tai lännessä
81  {
82  if(tileX > edgeX && tileOk(tileX+1,tileY) && tileOk(tileX+2,tileY)) // Vapaa tila idässä
83  {
84  spaces.emplace_back(tileX,tileY,tileX+1,tileY,tileX+2,tileY,mapGenerator::WALL_WEST);
85  map_->setTile(t,TileType::PARKSPACE_HORIZONTAL);
86  map_->setTile(tileX+1,tileY,TileType::NEXT_PARK_TILE);
87  } else if(tileOk(tileX-1,tileY) && tileOk(tileX-2,tileY)) // Vapaa tila lännessä
88  {
89  spaces.emplace_back(tileX,tileY,tileX-1,tileY,tileX-2,tileY,mapGenerator::WALL_EAST);
90  map_->setTile(t,TileType::NEXT_PARK_TILE);
91  map_->setTile(tileX-1,tileY,TileType::PARKSPACE_HORIZONTAL);
92  }
93  } else if(tileX == edgeX) { // seinä pohjoisessa tai etelässä
94  if(tileY > edgeY && tileOk(tileX,tileY-1) && tileOk(tileX,tileY-2)) // Vapaa tila pohjoisessa
95  {
96  spaces.emplace_back(tileX,tileY,tileX,tileY-1,tileX,tileY-2,mapGenerator::WALL_SOUTH);
97  map_->setTile(t,TileType::NEXT_PARK_TILE);
98  map_->setTile(tileX,tileY-1,TileType::PARKSPACE_VERTICAL);
99  } else if(tileOk(tileX,tileY+1) && tileOk(tileX,tileY+2)) //Vapaa tila etelässä
100  {
101  spaces.emplace_back(tileX,tileY,tileX,tileY+1,tileX,tileY+2,mapGenerator::WALL_NORTH);
102  map_->setTile(t,TileType::PARKSPACE_VERTICAL);
103  map_->setTile(tileX,tileY+1,TileType::NEXT_PARK_TILE);
104  }
105  }
106  }
107  /* Tarkistetaan että parkkiruudusta pääsee pois
108  * Tilanne:
109  * 1 2 3
110  * = = x
111  *
112  * Tarkistetaan x:n ylä- ja alapuolelta että siellä on tilaa
113  */
114  for(const auto &s : spaces)
115  {
116  if( !tileOk(s.nextX,s.nextY))
117  {
118  map_->setTile(s.x1,s.y1,TileType::FLOOR);
119  map_->setTile(s.x2,s.y2,TileType::FLOOR);
120  }
121  if( s.dir == mapGenerator::WALL_NORTH && (!tileOk(s.nextX-1,s.nextY) || !tileOk(s.nextX+1,s.nextY)))
122  {
123  map_->setTile(s.x1,s.y1,TileType::FLOOR);
124  map_->setTile(s.x2,s.y2,TileType::FLOOR);
125  } else if( s.dir == mapGenerator::WALL_EAST && (!tileOk(s.nextX,s.nextY-1) || !tileOk(s.nextX,s.nextY+1)))
126  {
127  map_->setTile(s.x1,s.y1,TileType::FLOOR);
128  map_->setTile(s.x2,s.y2,TileType::FLOOR);
129  } else if(s.dir == mapGenerator::WALL_SOUTH && (!tileOk(s.nextX-1,s.nextY) || !tileOk(s.nextX+1,s.nextY)))
130  {
131  map_->setTile(s.x1,s.y1,TileType::FLOOR);
132  map_->setTile(s.x2,s.y2,TileType::FLOOR);
133  } else if(s.dir == mapGenerator::WALL_WEST && (!tileOk(s.nextX,s.nextY-1) || !tileOk(s.nextX,s.nextY+1)))
134  {
135  map_->setTile(s.x1,s.y1,TileType::FLOOR);
136  map_->setTile(s.x2,s.y2,TileType::FLOOR);
137  }
138  }

162 void parkSpaceInserter::removeBlockedSpaces()
163 {
164  for(size_t i = 0; i < map_->tiles_.size(); ++i)
165  {
166  if(map_->getTile(i) == TileType::ACCESSIBLE)
167  {
168  map_->setTile(i,TileType::FLOOR);
169  }
170  }
171 
172  std::function<bool(int,int)> tileAccessible = [=](int x,int y) { return map_->getTile(x,y) == TileType::ACCESSIBLE; };
173 
174  int unaccessibleTiles = 0;
175  std::function<bool(int,int)> parkingSpace = [=](int x,int y) { return map_->getTile(x,y) == TileType::PARKSPACE_HORIZONTAL ||
176  map_->getTile(x,y) == TileType::PARKSPACE_VERTICAL || map_->getTile(x,y) == TileType::NEXT_PARK_TILE; };
177  int entryTile = map_->indexAt(map_->startX,map_->startY);
178  int iterations = 0;
179  // Iteroidaan, kunnes alusta pääsee kaikkiin ruutuihin
180  do {
181  unaccessibleTiles = 0;
182  floodFill(entryTile);
183  for(const auto &t : map_->tiles_)
184  {
185  if(t == TileType::FLOOR)
186  {
187  ++unaccessibleTiles;
188  }
189  }
190  if(unaccessibleTiles == 0)
191  {
192  for(auto &t : map_->tiles_)
193  {
194  if(t == TileType::ACCESSIBLE)
195  {
196  t = TileType::FLOOR;
197  }
198  }
199  return;
200  }
201 
202  for(size_t i = 0; i < map_->tiles_.size(); ++i)
203  {
204  if(map_->getTile(i) == TileType::FLOOR)
205  {
206  int tileX = map_->indexToX(i);
207  int tileY = map_->indexToY(i);
208  int nextX = 0;
209  int nextY = 0;
210  bool ok = false;
211  if(tileAccessible(tileX+1,tileY-1))
212  {
213  if(parkingSpace(tileX+1,tileY))
214  {
215  ok = true;
216  nextX = tileX+1; nextY = tileY;
217  } else if(parkingSpace(tileX,tileY-1))
218  {
219  ok = true;
220  nextX = tileX; nextY = tileY-1;
221  }
222  } else if(tileAccessible(tileX+1,tileY+1))
223  {
224  if(parkingSpace(tileX+1,tileY))
225  {
226  ok = true;
227  nextX = tileX+1; nextY = tileY;
228  } else if(parkingSpace(tileX,tileY+1))
229  {
230  ok = true;
231  nextX = tileX; nextY = tileY+1;
232  }
233  } else if(tileAccessible(tileX-1,tileY+1))
234  {
235  if(parkingSpace(tileX,tileY+1))
236  {
237  ok = true;
238  nextX = tileX; nextY = tileY+1;
239  } else if(parkingSpace(tileX-1,tileY))
240  {
241  ok = true;
242  nextX = tileX-1; nextY = tileY;
243  }
244  } else if(tileAccessible(tileX-1,tileY-1))
245  {
246  if(parkingSpace(tileX-1,tileY))
247  {
248  ok = true;
249  nextX = tileX-1; nextY = tileY;
250  } else if(parkingSpace(tileX,tileY-1))
251  {
252  ok = true;
253  nextX = tileX; nextY = tileY-1;
254  }
255  }
256  if(ok)
257  {
258  map_->setTile(tileX,tileY,TileType::ACCESSIBLE);
259  map_->setTile(nextX,nextY,TileType::ACCESSIBLE);
260  --unaccessibleTiles;
261  }
262  }
263  }
264 
265  } while(iterations++ < 6 && unaccessibleTiles > 0);

267 void parkSpaceInserter::floodFill(int tile)
268 {
269  static std::map<int, struct Direction > adjacentWallDirection =
270  { {0, {0,0,0}}, {1, {1,0,1}}, {2, {2,-1,0}}, {3, {3,0,-1}}, {4,{4,1,0}}};
271 
272  if(map_->getTile(tile) != TileType::FLOOR)
273  {
274  return;
275  }
276  // TileType::EXIT merkkaa, että ruutuun pääsee lähtöpaikasta
277  map_->setTile(tile,TileType::ACCESSIBLE);
278  for(auto dir : adjacentWallDirection)
279  {
280  struct Direction next = getRelativeDirection(map_->indexToX(tile),map_->indexToY(tile),dir.second);
281  floodFill(map_->indexAt(next.nextX,next.nextY));
282  }
283  return;
284 }
