87  }

92  int nextY;
93  };
94  std::map<int, struct Direction > adjacentWallDirection = { {0, {0,1}}, {2, {-1,0}}, {3, {0,-1}}, {4,{1,0}}};
95  while(!distanceQueue.empty())

103  for(auto dirPair : adjacentWallDirection)
104  {

107  if(nextX < 0 || nextX >= width || nextY < 0 || nextY >= height)

133 std::vector<int> mapProcessor::getMedialMap()
134 {
135  calculateDistanceMap();
136  // Medial map laskee etäisyysmuunnoksen perusteella kohtia, joissa etäisyys seinään on suurimmillaan
137  std::vector<int> medialMap = std::vector<int>(width*height,0);
138 
139  int counter = 0;
140 
141  size_t end = distanceMap.size();
142 
143  for(size_t t = 0; t < end; ++t)
144  {
145  int x = generator_->indexToX(counter);
146  int y = generator_->indexToY(counter);
147  if(generator_->getTile(x,y) == TileType::WALL || generator_->getTile(x,y) == TileType::EDGE)
148  {
149  counter++;
150  continue;
151  }
152  int closestTileIndex = getTileClosest(counter);
153 
154  // Käydään läpi parit tämä-pohjoinen, tämä-itä ja pohjoinen-itä
155  // Jokaiselle parille lasketaan etäisyys tilejen lähimpien seinien välillä
156  // MedialMap:n arvoksi laitetaan näiden etäisyyksien neliösumma
157  // Mikäli arvo on jotain raja-arvoa suurempi, on tile keskikohta
158  int northX = x;
159  int northY = y-1;
160  if(generator_->getTile(northX,northY) != TileType::FLOOR)
161  {
162  counter++;
163  continue;
164  }
165  int northClosestIndex = getTileClosest(northX,northY);
166 
167  int eastX = x+1;
168  int eastY = y;
169  if(generator_->getTile(eastX,eastY) != TileType::FLOOR)
170  {
171  counter++;
172  continue;
173  }
174 
175  int eastClosestIndex = getTileClosest(eastX,eastY);
176 
177  int closestNorth = getDistance(closestTileIndex,northClosestIndex);
178  int closestEast = getDistance(closestTileIndex,eastClosestIndex);
179  int eastNorth = getDistance(eastClosestIndex,northClosestIndex);
180 
181  closestNorth*=closestNorth;
182  closestEast*=closestEast;
183  eastNorth*=eastNorth;
184  medialMap[counter] = /*1*/ closestNorth+closestEast+eastNorth;
185  ++counter;
186  }
187  return medialMap;
