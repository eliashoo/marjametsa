212 // void ActionScenePC::obstacleHit(bool topRight, bool bottomRight,
213 // bool bottomLeft, bool topLeft)
214 // ============================================================================
215 // Jos hahmo on osunut esteeseen, siirtää hahmoa esteestä poispäin.
216 //
217 // Parametrit:
218 // (1) topRight Este yläoikealla.
219 // (2) bottomRight Este alaoikealla.
220 // (3) bottomLeft Este alavasemmalla.
221 // (4) topLeft Este ylävasemmalla.
222 //
223 // Ei paluuarvoja.
224 // ============================================================================
225 void ActionScenePC::obstacleHit(bool topRight, bool bottomRight,
226  bool bottomLeft, bool topLeft)
227 {
228  int dx{0};
229  int dy{0};
230 
231  int displacement{10};
232 
233  if( topRight ) {
234  dx -= displacement;
235  dy += displacement;
236  }
237  if( bottomRight ) {
238  dx -= displacement;
239  dy -= displacement;
240  }
241  if( bottomLeft ) {
242  dx += displacement;
243  dy -= displacement;
244  }
245  if( topLeft ) {
246  dx += displacement;
247  dy += displacement;
248  }
249 
250  // Jos siirtymäksi tulee yli haluttu siirtymä, asetetaan se haluttuun, mutta
251  // pidetään suunta.
252  if( dx > displacement ) {
253  dx = displacement;
254  }
255  if( dx < -displacement ) {
256  dx = -displacement;
257  }
258  if( dy > displacement ) {
259  dy = displacement;
260  }
261  if( dy < -displacement ) {
262  dy = -displacement;
263  }
264 
265  setPos(x()+dx,y()+dy);
266  muzzleFlash_->setPos(x(),y());
267 
268  // Pysäytetään liike esteeseen osuttaessa.
269  moveTimer_->stop();
270 }

614  moveTimer_->stop();
