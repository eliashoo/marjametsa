304 // bool PlayerParty::reloadWeapon(unsigned int ID, unsigned int weaponSlot)
305 // ============================================================================
306 // Tätä funktiota käytetään, kun actori tai apuri lataa aseen. Laskeetaan
307 // latauksen arvo [(nykyinen ammusmäärä / maksimimäärä) * (aseen hinta / 2)].
308 //
309 // Parametrit:
310 // (1) unsigned int ID: Actorin ID.
311 // (2) unsigned int weaponSlot: Aseen slotti.
312 //
313 // Paluuarvot:
314 // (1) bool true: Lataus onnistui tai ei tarvinnut ladata.
315 // (2) bool false: Ei onnistunut.
316 // ============================================================================
317 bool PlayerParty::reloadWeapon(unsigned int ID, unsigned int weaponSlot)
318 {
319  unsigned int currentAmmo{getActor(ID)->getWeapon(weaponSlot).getAmmo()};
320  unsigned int maxAmmo{getActor(ID)->getWeapon(weaponSlot).getMaximumAmmoCapacity()};
321  double money{double(getActor(0)->getMoney())};
322  unsigned int ammoToReload{maxAmmo - currentAmmo};
323 
324  if ( currentAmmo == maxAmmo ) {
325  qDebug() << "Weapon is already full.";
326  return true;
327  }
328  double percentage{2.0 - (double(currentAmmo)/double(maxAmmo))};
329  double price{double(getActor(ID)->getWeapon(weaponSlot).getPrice())};
330  price = ((price / 2) * percentage);
331 
332  if ( money < price ) {
333  percentage = double(money) / double (price);
334  ammoToReload = std::round(ammoToReload * percentage);
335  price = money;
336  }
337  qDebug() << "Reload price: [" << price << "]";
338  getActor(ID)->reloadWeapon(weaponSlot, ammoToReload);
339  std::shared_ptr<Interface::ActorInterface> tempActor{getActor(0)};
340  auto actor = std::dynamic_pointer_cast<Player>(tempActor);
341  actor->reduceMoney(std::round(price));
342  return true;
343 }
